def self.all
  config  = Configuration.new
  suites  = Suite.all
  dirs    = []
  rootdir = Cfg.rootdir
  suites.each do |suite|
    components = Dir.glob(File.join(rootdir, suite, "*"))
    components.each do |component|
      dirs.push component.gsub(%r#{rootdir}\//, '') if File.directory? component
    end
  end
  return dirs
end

def self.dataset(category=nil)
  data  = []
  self.all.each do |entry|
    parts = entry.split(%r\//)
    unless parts.length < 3
      next unless parts[0].eql?(category) || category.eql?("all")
      data << {
        :category     => parts[0],
        :suitename    => parts[1],
        :component    => parts[2],
        :fullpath     => File.join(Cfg.rootdir, entry)
      }
    end
  end
  data
end

def is_unused?(dir)
  status  = true

  path = Dir.glob(File.join(dir, "*"))
  path.each do |dirorfile|
    status = false if File.directory?(dirorfile)
    status = false if File.basename(dirorfile) =~ %r\.deb$/
  end

  status
end
