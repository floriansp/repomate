def self.dataset(category=nil)
  data   = []
  self.all.each do |entry|
    unless entry.nil?
      next unless entry.eql?(category) || category.eql?("all")
      data << {
        :category     => entry,
        :fullpath     => File.join(Cfg.rootdir, entry)
      }
    end
  end
  data
end
