 93:           target_package = Package.new(target_fullname, entry[:suitename], entry[:component] )
 94: 
 95:           if system("#{dpkg} --compare-versions #{source_package.version} gt #{target_package.version}")
 96:             puts "Package: #{target_package.newbasename} will be replaced with #{source_package.newbasename}"
 97:             unlink_workload << {
 98:               :destination_fullname => target_fullname,
 99:               :newbasename          => target_package.newbasename
100:             }
101:           elsif system("#{dpkg} --compare-versions #{source_package.version} eq #{target_package.version}")
102:             puts "Package: #{source_package.newbasename} already exists with same version"
103:             return
104:           elsif system("#{dpkg} --compare-versions #{source_package.version} lt #{target_package.version}")
105:             puts "Package: #{source_package.newbasename} already exists with higher version"
106:             return
107:           end
108:         end
109: 
110:         link_workload << {
111:           :source_fullname      => source_package.fullname,
112:           :destination_fullname => destination_fullname,
113:           :suitename            => source_package.suitename,
114:           :component            => source_package.component,
115:           :newbasename          => source_package.newbasename
