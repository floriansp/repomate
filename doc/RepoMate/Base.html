 93:     def link(workload)
 94:       dpkg   = @config.get[:dpkg]
 95: 
 96:       raise "dpkg is not installed" unless File.exists?(dpkg)
 97: 
 98:       link   = []
 99:       unlink = []
100:       action = false
101: 
102:       workload.each do |entry|
103:         source_package       = Package.new(entry[:source_fullname], entry[:suitename], entry[:component])
104:         destination_fullname = File.join(entry[:destination_dir], source_package.newbasename)
105: 
106:         Dir.glob("#{entry[:destination_dir]}/#{source_package.name}*.deb") do |target_fullname|
107:           target_package = Package.new(destination_fullname, entry[:suitename], entry[:component] )
108: 
109:          if system("#{dpkg} --compare-versions #{source_package.version} gt #{target_package.version}")
110:             puts "Package: #{target_package.newbasename} will be replaced with #{source_package.newbasename}"
111:             unlink << {
112:               :destination_fullname => target_fullname,
113:               :newbasename          => target_package.newbasename
114:             }
115:           elsif system("#{dpkg} --compare-versions #{source_package.version} eq #{target_package.version}")
116:           puts "Package: #{source_package.newbasename} already exists with same version"
117:           elsif system("#{dpkg} --compare-versions #{source_package.version} lt #{target_package.version}")
118:           puts "Package: #{source_package.newbasename} already exists with higher version"
119:           end
120:         end
121: 
122:         link << {
123:           :source_fullname      => entry[:source_fullname],
124:           :destination_fullname => destination_fullname,
125:           :suitename            => entry[:suitename],
126:           :component            => entry[:component],
127:           :newbasename          => source_package.newbasename
128:         }
129:       end
130: 
131:       unlink.each do |entry|
132:         File.unlink(entry[:destination_fullname])
133:         puts "Package: #{entry[:newbasename]} unlinked"
134:         action = true
135:       end
136: 
137:       link.each do |entry|
138:         unless File.exists?(entry[:destination_fullname])
139:           File.symlink(entry[:source_fullname], entry[:destination_fullname])
140:           puts "Package: #{entry[:newbasename]} linked to production => #{entry[:suitename]}/#{entry[:component]}"
