Does the link job after checking versions through dpkg 

     # File lib/repomate/base.rb, line 92
 92:     def link(workload)
 93:       dpkg   = @config.get[:dpkg]
 94: 
 95:       raise "dpkg is not installed" unless File.exists?(dpkg)
 96: 
 97:       link   = []
 98:       unlink = []
 99:       action = false
100: 
101:       workload.each do |entry|
102:         @repository.create(entry[:suitename], entry[:component], entry[:architecture])
103: 
104:         source_package       = Package.new(entry[:source_fullname], entry[:suitename], entry[:component])
105:         destination_fullname = File.join(entry[:destination_dir], source_package.newbasename)
106: 
107:         Dir.glob("#{entry[:destination_dir]}/#{source_package.name}*.deb") do |target_fullname|
108:           target_package = Package.new(destination_fullname, entry[:suitename], entry[:component] )
109: 
110:          if system("#{dpkg} --compare-versions #{source_package.version} gt #{target_package.version}")
111:             puts "Package: #{target_package.newbasename} will be replaced with #{source_package.newbasename}"
112:             unlink << {
113:               :destination_fullname => target_fullname,
114:               :newbasename          => target_package.newbasename
115:             }
116:           elsif system("#{dpkg} --compare-versions #{source_package.version} eq #{target_package.version}")
117:             puts "Package: #{source_package.newbasename} already exists with same version"
118:           elsif system("#{dpkg} --compare-versions #{source_package.version} lt #{target_package.version}")
119:             puts "Package: #{source_package.newbasename} already exists with higher version"
120:           end
121:         end
122: 
123:         link << {
124:           :source_fullname      => entry[:source_fullname],
125:           :destination_fullname => destination_fullname,
126:           :suitename            => entry[:suitename],
127:           :component            => entry[:component],
128:           :newbasename          => source_package.newbasename
129:         }
130:       end
131: 
132:       unlink.each do |entry|
133:         File.unlink(entry[:destination_fullname])
134:         puts "Package: #{entry[:newbasename]} unlinked"
135:         action = true
136:       end
137: 
138:       link.each do |entry|
139:         unless File.exists?(entry[:destination_fullname])
140:           File.symlink(entry[:source_fullname], entry[:destination_fullname])
141:           puts "Package: #{entry[:newbasename]} linked to production => #{entry[:suitename]}/#{entry[:component]}"
142:           action = true
143:         end
144:       end
145: 
146:       if action
147:         cleandirs
148:         @metafile.create
149:       end
150:     end
